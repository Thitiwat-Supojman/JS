ให้คุณใช้ Multithreading เพื่อประมวลผล 2 งานพร้อมกัน Task task1 = Task.Run(() => PrintNumbers("Task 1")); Task task2 = Task.Run(() => PrintNumbers("Task 2")); Task.WaitAll(task1, task2); // รอให้ Task ทั้งหมดทำงานเสร็จ static void PrintNumbers(string taskName) { for (int i = 1; i <= 5; i++) { Console.WriteLine($"{taskName}: {i}"); Thread.Sleep(500); // จำลองการทำงานหนัก } }
