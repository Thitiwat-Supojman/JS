initialBalance : 0; } // เมทอดสำหรับฝากเงินเข้าบัญชี public void Deposit(decimal amount) { if (amount > 0) balance += amount; // เงินฝากต้องมากกว่า 0 } // เมทอดสำหรับถอนเงินจากบัญชี public bool Withdraw(decimal amount) { if (amount > 0 && amount <= balance) // ตรวจสอบว่าเงินที่ถอนต้องไม่น้อยกว่าหรือมากกว่ายอดเงินในบัญชี { balance -= amount; return true; // ถอนสำเร็จ } return false; // ถอนล้มเหลว (เงินไม่พอ) } // เมทอดสำหรับเช็คยอดเงินในบัญชี public decimal GetBalance() => balance; // เมทอดแสดงรายละเอียดบัญชี public override string ToString() { return $"Account: {AccountNumber}, Owner: {Owner}, Balance: {balance:C}"; } } class Bank { private List accounts = new List(); // เก็บรายการบัญชีทั้งหมด // เพิ่มบัญชีใหม่เข้าระบบธนาคาร public void AddAccount(BankAccount account) => accounts.Add(account); // ค้นหาบัญชีตามหมายเลขบัญชี public BankAccount GetAccount(string accountNumber) => accounts.FirstOrDefault(a => a.AccountNumber == accountNumber); // คำนวณยอดรวมของเงินทั้งหมดในธนาคาร public decimal GetTotalBalance() => accounts.Sum(a => a.GetBalance()); // ค้นหาบัญชีที่มียอดเงินอยู่ในช่วงที่กำหนด public List GetAccountsByBalance(decimal min, decimal max) => accounts.Where(a => a.GetBalance() >= min && a.GetBalance() <= max).ToList(); }
